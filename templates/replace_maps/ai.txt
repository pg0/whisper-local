# AI integration samples (off by default — enable in the tray).
#
# Every trigger takes the current selection on stdin and types the reply
# back over the selection. Oneliners are active by default; helper-script
# versions sit right next to them as comments for when you want to edit
# model / endpoint / system prompt in a file.
#
# Helpers live in %APPDATA%\whisper-local\helpers\ (both .ps1 and .py ship).

# ---------------------------------------------------------------- env vars
# $env:OPENAI_API_KEY      = 'sk-...'
# $env:OPENROUTER_API_KEY  = 'sk-or-...'
# setx OPENAI_API_KEY "sk-..."     (persist for your user, new terminals)

# =============================================================== Claude CLI
# Anthropic's `claude` must be on PATH.
#
# >>exec: rewrites the SELECTION (select text first, then speak trigger).
# >>cmd:  runs without stdin; used with regex captures so "ask claude <prompt>"
#         types the answer wherever the caret is.

ask claude:>>exec:powershell -NoProfile -Command "$i=[Console]::In.ReadToEnd(); $i | claude -p --allowedTools 'Read,Edit,Bash' -"
/^ask claude (.+)$/i:>>cmd:powershell -NoProfile -Command "'$1' | claude -p --allowedTools 'Read,Edit,Bash' -"

# =============================================================== OpenAI
ask gpt:>>exec:powershell -NoProfile -Command "$i=[Console]::In.ReadToEnd(); $b=@{model='gpt-4o-mini';messages=@(@{role='user';content=$i})}|ConvertTo-Json -Depth 8 -Compress; (Invoke-RestMethod -Uri 'https://api.openai.com/v1/chat/completions' -Method Post -Headers @{Authorization=\"Bearer $env:OPENAI_API_KEY\"} -ContentType 'application/json' -Body ([Text.Encoding]::UTF8.GetBytes($b))).choices[0].message.content"
/^ask gpt (.+)$/i:>>cmd:powershell -NoProfile -Command "$b=@{model='gpt-4o-mini';messages=@(@{role='user';content='$1'})}|ConvertTo-Json -Depth 8 -Compress; (Invoke-RestMethod -Uri 'https://api.openai.com/v1/chat/completions' -Method Post -Headers @{Authorization=\"Bearer $env:OPENAI_API_KEY\"} -ContentType 'application/json' -Body ([Text.Encoding]::UTF8.GetBytes($b))).choices[0].message.content"
